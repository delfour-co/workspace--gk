Sieve Quick Reference

Common Conditions

# Match header content
header :contains "Subject" "newsletter"

# Match sender domain
address :domain "From" "example.com"

# Match message size
size :over 5M

# Combine conditions
allof (
  header :contains "From" "boss",
  header :contains "Subject" "urgent"
)

Common Actions

# File into folder
fileinto "Newsletters";

# Discard message
discard;

# Redirect to another address
redirect "other@example.com";

# Keep message (default)
keep;

# Stop processing
stop;

Example Script

require "fileinto";

# File newsletters into separate folder
if header :contains "Subject" "newsletter" {
    fileinto "Newsletters";
    stop;
}

# Discard spam from known sender
if address :is "From" "spam@example.com" {
    discard;
    stop;
}

# Move large attachments to separate folder
if size :over 10M {
    fileinto "Large Files";
}
